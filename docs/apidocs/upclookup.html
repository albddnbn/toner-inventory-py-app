154            db_name=self.dbname,

156            dicttrue=True,
157        )

177                # I don't think this try catch is necessary, but keep for now
178                except:
179                    return False

181    def parse_info(self, data:dict, upc:str) -> tuple:
182        """
183        parse_info Parse information from a json response file from upc lookup API.
184
185        Args:
186            data (dictionary): json data from upc lookup API
187            upc (str): upc code of the item
188
189        Returns:
190            tuple: important information about item including name, item type, color, etc.
191        """
192        # get info dictionary from the items key (to see example of what resulting json looks like, check json directory)
193        info = data['items'][0]
194        item_name = info['title']
195        name_split = item_name.split(" ")
196        name_split = [x.lower() for x in name_split]
197
198        # item name/description
199        item_desc = info['description']
200        split_desc = item_desc.split(" ")
201        split_desc = [word.lower() for word in split_desc]
202
203        # manufacturer
204        try:
205            item_brand = info['brand'].lower()
206        except:
207            item_brand = ""
208        # if item brand isn't already one of the major ones, then do some further checks on the name and description
209        if item_brand not in ["dell", "hp", "lexmark", "sharp", "canon", "epson"]:
210            if item_brand == "hewlett packard":
211                item_brand = "hp"
212            else:
213                # check if dell/hp/lexmark is in the name or description
214                for manufacturer in ["dell", "hp", "lexmark", "epson", "canon", "hewlett"]:
215                    if manufacturer in name_split:
216                        item_brand = manufacturer
217                    else:
218                        if manufacturer in split_desc:
219                            item_brand = manufacturer
220
221        item_code = info['model']
222
223        item_color = self.parse_color(info, upc)
224        item_type = self.parse_type(info, upc)
225
226        # chop description down a bit if its too long - already storing the whole thing in the json file
227        if len(item_desc) >= 250:
228            # chop off first 250 characters of the item description, split it up by spaces (into words), then chop off the last word in case it's cut off
229            item_desc = item_desc[:249].split(" ")[:-1]
230            # join the item description list back together into a string
231            item_desc = " ".join(item_desc)
232
233        # do basically the same with item names
234        split_name = item_name.split(" ")
235        characters = 0
236        item_name = ""
237        for word in split_name:
238            if (characters <= 250) and (characters + len(word) <= 250):
239                characters += len(word)
240                item_name += f" {word}"
241
242        # return variables:
243        return upc, item_brand, item_name, item_desc, item_code, item_color, item_type

300    def parse_type(self, info_data:dict, upc:str) -> str:
301        """
302        parse_type Attempts to parse the type of the item from title and description if not explicitly listed in the type property of the json from api lookup.
303
304        Args:
305            info_data (dict): chunk of data from the json file from the api lookup
306            upc (str): upc code of the item

315        # GET THE ITEM TYPE:
316        for itype in ["toner", "ink", "drum", "imaging", "fuser", "waste"]:
317            if itype in split_title:
318                # if its on "imaging" iteration of the for loop and it pops, check for the word "unit" too before assigning it to imaging unit type
319                if itype == "imaging":
320                    # if "unit" in split_title: --> this would check for the word 'unit' as well as 'imaging'
321                    item_type = "drum"
322                else:
323                    item_type = itype
324            # if it cant find the item type in the name, check description (this may not work perfectly)
325            else:
326                if itype in split_desc:
327                    if itype == "imaging":
328                        item_type = "drum"
329                    else:
330                        item_type = itype
331        return item_type

183        """
184        parse_info Parse information from a json response file from upc lookup API.
185
186        Args:
187            data (dictionary): json data from upc lookup API
188            upc (str): upc code of the item
189
190        Returns:
191            tuple: important information about item including name, item type, color, etc.
192        """
193        # get info dictionary from the items key (to see example of what resulting json looks like, check json directory)
194        info = data['items'][0]
195        item_name = info['title']
196        name_split = item_name.split(" ")
197        name_split = [x.lower() for x in name_split]
198
199        # item name/description
200        item_desc = info['description']
201        split_desc = item_desc.split(" ")
202        split_desc = [word.lower() for word in split_desc]
203
204        # manufacturer
205        try:
206            item_brand = info['brand'].lower()
207        except:
208            item_brand = ""
209        # if item brand isn't already one of the major ones, then do some further checks on the name and description
210        if item_brand not in ["dell", "hp", "lexmark", "sharp", "canon", "epson"]:
211            if item_brand == "hewlett packard":
212                item_brand = "hp"
213            else:
214                # check if dell/hp/lexmark is in the name or description
215                for manufacturer in ["dell", "hp", "lexmark", "epson", "canon", "hewlett"]:
216                    if manufacturer in name_split:
217                        item_brand = manufacturer
218                    else:
219                        if manufacturer in split_desc:
220                            item_brand = manufacturer
221
222        item_code = info['model']
223
224        item_color = self.parse_color(info, upc)
225        item_type = self.parse_type(info, upc)
226
227        # chop description down a bit if its too long - already storing the whole thing in the json file
228        if len(item_desc) >= 250:
229            # chop off first 250 characters of the item description, split it up by spaces (into words), then chop off the last word in case it's cut off
230            item_desc = item_desc[:249].split(" ")[:-1]
231            # join the item description list back together into a string
232            item_desc = " ".join(item_desc)
233
234        # do basically the same with item names
235        split_name = item_name.split(" ")
236        characters = 0
237        item_name = ""
238        for word in split_name:
239            if (characters <= 250) and (characters + len(word) <= 250):
240                characters += len(word)
241                item_name += f" {word}"
242
243        # return variables:
244        return upc, item_brand, item_name, item_desc, item_code, item_color, item_type
