216    def get_table_data(self, query:str):
217
218        db, cursor = connect_to_db(
219            db_host=self.dbhost,
220            db_user=self.dbuser,
221            db_name=self.dbname,
222            db_pass=self.dbpass,
223            dicttrue=False,
224        )
225        cursor.execute(query)
226        results = cursor.fetchall()
227        cursor.close()
228        db.close()
229        # convert results to be used as model
230        table_rows = []
231        for inner_tuple in results:
232            item_details = []
233            for detail in inner_tuple:
234                item_details.append(detail)
235            table_rows.append(item_details)
236        
237        table_rows = sorted(table_rows, key=lambda x: x[1])
238        return table_rows

261    def headerData(self, section, orientation, role):
262        if role == Qt.DisplayRole:
263            if orientation == Qt.Horizontal:
264                return self.headers[section]

266    def data(self, index, role):
267        row = index.row()
268        col = index.column()
269
270        column_name = self.headerData(
271            col, Qt.Horizontal, Qt.DisplayRole)
272        value = self.modeldata[row][col]
273        if role == Qt.DisplayRole:
274            if column_name.lower() in ["stock"]:
275                return QVariant(value)
276            if column_name.lower() == "t":
277                return ""
278            return QVariant(value)
279        elif role == Qt.TextAlignmentRole:
280            if (col in [3, 4, 5, 6, 9]) or column_name.lower() in ["stock", "upc", "type", "needed"]:
281                return QVariant(Qt.AlignCenter)
282        elif role == Qt.DecorationRole and column_name.lower() == "t":
283            if self.modeldata[row][col] in ["ink", "toner"]:
284                toner_icon = qtawesome.icon(
285                    "ri.paint-fill", color="#0252c2", size=56)
286                return QVariant(toner_icon)
287            elif self.modeldata[row][col] in ["paper"]:
288                paper_icon = qtawesome.icon(
289                    "ri.file-paper-2-fill", color="#0252c2")
290                return QVariant(paper_icon)
291            elif self.modeldata[row][col] in ["drum", "fuser"]:
292                drum_icon = qtawesome.icon(
293                    "ri.printer-line", color="#0252c2", size=56)
294                return QVariant(drum_icon)
295            else:
296                other_icon = qtawesome.icon(
297                    "ri.printer-line", color="#0252c2", size=56)
298                return QVariant(other_icon)
299        elif role == Qt.FontRole and value == 0:
300            font = QFont('Roboto', 12, QFont.Bold)
301            font.setBold(True)
302            font.setWeight(75)
303            return QVariant(font)
304        elif role == Qt.FontRole:
305            font = QFont('Roboto', 12)
306            font.setBold(False)
307            font.setWeight(50)
308            return QVariant(font)
309
310        elif role == Qt.BackgroundRole:
311            if column_name.lower() == "stock":
312                try:
313                    value = int(value)  # Convert to integer for indexing.
314                    value = min(4, value)
315
316                    return QVariant(QColor(self.colorset[value]))
317                except ValueError:
318                    pass
319        return QVariant()
